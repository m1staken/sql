-- 3. Напишите команду, которая бы увеличила количество книг «Первая» на 10 штук в заказе для заказчика из Ростова.
UPDATE Items
SET B_COUNT = B_COUNT + 10
WHERE 
	B_ID IN (
        SELECT B_CONTRACT
        FROM Books
        WHERE B_TITLE = '«Первая»'
    )
	AND O_ID IN (
		SELECT O_ID 
		FROM Orders o 
		JOIN Customers c ON c.C_ID = o.O_COMPANY
		WHERE c.C_ADDR = 'Ростов');

-- 8. С помощью запроса заполняем поле B_REST (остаток тиража = тираж – сумма количества книг в заказах) в таблице Books (Книги) для книги «Первая».
UPDATE Books
SET B_REST = B_CIRCUL - (
    SELECT (SUM(i.B_COUNT))
    FROM Items i
    WHERE i.B_ID = Books.B_CONTRACT
)
WHERE B_TITLE = '«Первая»';
