-- 5. Создайте представление для просмотра названий всех книг отредактированных указанным редактором Андреев Александр

CREATE VIEW Fifth AS
SELECT B_TITLE
FROM Books b
JOIN Editors e ON e.B_ID = b.B_CONTRACT
JOIN Employees em ON em.E_ID = e.E_ID 
WHERE em.E_FNAME = 'Андреев' AND em.E_LNAME = 'Александр';
