-- ДОПЗАПРОС: Вывести всех сотрудников которые работают в должности, заданной через параметр

CREATE PROCEDURE Procedure7
	@E_Post NVARCHAR(40)
AS
BEGIN
	SELECT E_FNAME
	FROM Employees em
	JOIN Posts p ON p.P_ID = em.E_POST
	WHERE @E_Post = p.P_POST;
END;
